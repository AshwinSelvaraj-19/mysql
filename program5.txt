CREATE TABLE STUDENT (
Name varchar (35),
Age INT,
Score INT,
Grade CHAR (10)
);
CREATE TRIGGER sample_trigger
BEFORE INSERT ON STUDENT
FOR EACH ROW
BEGIN
IF :NEW.Score < 35 THEN
 :NEW.Grade := 'FAIL';
ELSE
 :NEW.Grade := 'PASS';
END IF;
END;
/
